# 字符串转换整数（atoi）
链接：https://leetcode.cn/problems/string-to-integer-atoi/?envType=problem-list-v2&envId=2ckc81c
```cpp
class Solution {
public:
    int myAtoi(string s) {
        int n=s.size();
        int i=0;
        while(i<n && s[i] ==' '){
            i++;
        }
        int sign=1;
        if(i<n && (s[i]=='+' || s[i] == '-')){
            sign= s[i]=='+'?  1:-1;
            i++;
        }
        int num=0;
        for(;i<n&& isdigit(s[i]);i++){
            int d= s[i]-'0';
            if(num> INT_MAX/10 || num*10 >INT_MAX-d){
                return sign>0 ? INT_MAX:INT_MIN;
            }
            num=num*10+d;
        }
        return sign*num;
        
    }
};

```

# 括号生成
链接： https://leetcode.cn/problems/generate-parentheses/description/?envType=problem-list-v2&envId=2ckc81c
```cpp
class Solution {
public:
    vector<string> generateParenthesis(int n) {
        int m=2*n;
        vector<string> ans;
        string path(m,' ');
        auto dfs=[&](this auto && dfs,int left,int right)->void{
            if(right==n){
                ans.push_back(path);
                return;
            }
            if(left<n){
                path[left+right]='(';
                dfs(left+1,right);
            }
            if(right<left){
                path[left+right]=')';
                dfs(left,right+1);
            }

        };
        dfs(0,0);
        return ans;
        
    }
};

```
