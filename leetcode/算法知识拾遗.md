# C++基础语法
# vector
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    // 创建并初始化
    vector<int> scores = {85, 92, 78, 90, 88};
    
    // 添加元素
    scores.push_back(95);
    
    // 遍历和计算
    int sum = 0;
    for (int score : scores) {
        sum += score;
    }
    double average = static_cast<double>(sum) / scores.size();
    cout << "平均分: " << average << endl;
    
    // 排序
    sort(scores.begin(), scores.end());
    cout << "最高分: " << scores.back() << endl;
    
    // 删除不及格分数
    scores.erase(
        remove_if(scores.begin(), scores.end(), 
                 [](int s){ return s < 60; }),
        scores.end()
    );
    
    return 0;
}
```

## 哈希表
常用的是unordered_map，，unordered_map 会自动处理哈希冲突
```cpp
#include <iostream>
#include <unordered_map>
#include <string>
using namespace std;

int main() {
    // 创建 unordered_map
    unordered_map<string, int> ageMap;
    
    // 插入元素
    ageMap["Alice"] = 25;
    ageMap["Bob"] = 30;
    ageMap["Charlie"] = 35;
    
    // 使用 insert
    ageMap.insert({"David", 28});
    ageMap.emplace("Eve", 32);  // 原地构造，更高效
    
    // 访问元素
    cout << "Alice's age: " << ageMap["Alice"] << endl;
    cout << "Bob's age: " << ageMap.at("Bob") << endl;  // 会检查键是否存在
    
    // 遍历
    for (const auto& pair : ageMap) {
        cout << pair.first << ": " << pair.second << endl;
    }
    
    return 0;
}

```


```cpp
#include <iostream>
#include <unordered_set>
#include <vector>
using namespace std;

int main() {
    // 创建 unordered_set
    unordered_set<int> numSet = {1, 2, 3, 4, 5};
    
    // 插入元素
    numSet.insert(6);
    numSet.emplace(7);
    
    // 检查元素是否存在
    if (numSet.find(3) != numSet.end()) {
        cout << "3 is in the set" << endl;
    }
    
    // 遍历
    for (int num : numSet) {
        cout << num << " ";
    }
    cout << endl;
    
    return 0;
}
```

## 字符串
string类型
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    // 1. 空字符串
    string s1;
    
    // 2. 从C风格字符串初始化
    string s2 = "Hello World";
    string s3("Hello World");
    
    // 3. 重复字符
    string s4(5, 'A');           // "AAAAA"
    
    // 4. 从另一个string初始化
    string s5 = s2;               // 拷贝构造
    string s6(s2);                // 同上
    
    // 5. 从部分字符串初始化
    string s7("Hello World", 5);  // 前5个字符: "Hello"
    string s8(s2, 6);             // 从索引6开始: "World"
    string s9(s2, 6, 5);          // 从索引6开始，取5个字符: "World"
    
    // 6. 使用初始化列表 (C++11)
    string s10 = {'H', 'e', 'l', 'l', 'o'};
    string s11{'H', 'i'};
    
    return 0;
}


#include <iostream>
#include <string>
#include <cctype>  // 包含 isalnum, tolower, isupper 等函数
using namespace std;

int main() {
    string s = "Hello World! 123";
    
    // 1. 将大写字母转换为小写
    for (char& c : s) {
        c = tolower(c);
    }
    cout << "转换为小写: " << s << endl;  // 输出: hello world! 123
    
    // 2. 判断字符是否是字母数字
    for (char c : s) {
        if (isalnum(c)) {
            cout << "'" << c << "' 是字母数字字符" << endl;
        } else {
            cout << "'" << c << "' 不是字母数字字符" << endl;
        }
    }
    
    return 0;
}
```
